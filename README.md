## Getting started

```sh
# Create the hasura migrations to load into k8s hasura
tar cvfj ./nhost/metadata.tar.bz2 -C ../../deploy/deepserver-complete/nhost/nhost/nhost/metadata .
tar cvfj ./nhost/migrations.tar.bz2 -C ../../deploy/deepserver-complete/nhost/nhost/nhost/migrations default
```

```sh
helm dependency update nhost

# Install Traefik Resource Definitions:
kubectl apply -f https://raw.githubusercontent.com/traefik/traefik/v2.9/docs/content/reference/dynamic-configuration/kubernetes-crd-definition-v1.yml

# Install RBAC for Traefik:
kubectl apply -f https://raw.githubusercontent.com/traefik/traefik/v2.9/docs/content/reference/dynamic-configuration/kubernetes-crd-rbac.yml

helm repo add jetstack https://charts.jetstack.io
helm repo update

# the command below takes a while to finish:
helm install cert-manager jetstack/cert-manager --namespace default --version v1.11.0 --set installCRDs=true

kubectl apply -f nhost/mkcertissuer.yaml


helm install --debug nhost nhost > out.yaml
watch -n 0.5 kubectl get pods --all-namespaces
```
